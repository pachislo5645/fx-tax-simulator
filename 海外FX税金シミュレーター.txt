  <script>
    function calculateTax() {
      const profit = parseFloat(document.getElementById('profit').value);
      if (isNaN(profit) || profit < 0) {
        document.getElementById('result').innerText = "正しい金額を入力してください。";
        return;
      }

      const taxRate = 0.20 + 0.0215; // 20% 所得税 + 2.15% 復興特別所得税
      const taxAmount = Math.floor(profit * taxRate);

      document.getElementById('result').innerText = `概算税額：${taxAmount.toLocaleString()}円`;
    }
  </script>
